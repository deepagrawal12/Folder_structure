/**
  * @function luhn_checksum: Implementation of Luhn algorithm for SIM card validation
  * @param simVal
  */
function luhn_checksum(simVal) {
  const len = simVal.length;
  const parity = len % 2;
  let sum = 0;
  for (let i = len - 1; i >= 0; i--) {
    let d = parseInt(simVal.charAt(i));
    if (i % 2 === parity) { d *= 2; }
    if (d > 9) { d -= 9; }
    sum += d;
  }
  return sum % 10;
}
